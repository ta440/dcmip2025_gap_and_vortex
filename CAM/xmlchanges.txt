# To record the xmlchanges for the gap flow and vortex shedding tests.
# These use a small earth factor of X=20.

# Avoid restart files
./xmlchange DOUT_S=FALSE,\
REST_OPTION=none,\
STOP_OPTION=nsteps
  
# Physics timestep for 0.5 degree resolution and X=20
./xmlchange ATM_NCPL=1920

# For FV3 only: change NTASKS; default CAM choice doesn't work
./xmlchange NTASKS=768

# Set NTASKS for SE and MPAS
./xmlchange NTASKS=512

# For the gap flow, equivalent of 10 standard Earth days
./xmlchange STOP_N=960 

# For the vortex street, the equivalent of 20 standard Earth days
./xmlchange STOP_N=1920

# Use the Held-Suarez compset to enable modified Rayleigh Friction
# and use the L57 20 km stretched grid.
./xmlchange --file env_build.xml --id CAM_CONFIG_OPTS --val \
"--phys held_suarez --analytic_ic --nlev=57"

./xmlchange JOB_WALLCLOCK_TIME=01:00:00
